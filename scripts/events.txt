import crafttweaker.event.PlayerPickupItemEvent;

    global damagingPickupItems as IItemStack[] = [
        <modid:item>,
        <modid:other_item>
    ];

    events.onPlayerPickupItem(function(event as PlayerPickupItemEvent) {
        if (damagingPickupItems has event.item) {
            event.player.attackEntityFrom(IDamageSource.ON_FIRE(), 1.0f);
        }
    });
